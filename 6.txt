import 'dart:io';

void main() {
  stdout.write("Enter a string: ");
  String inputStr = stdin.readLineSync()!.toLowerCase();

  String cleanedStr = inputStr.replaceAll(' ', '');

  String reversedStr = cleanedStr.split('').reversed.join('');

  if (cleanedStr == reversedStr) {
    print("The string '$inputStr' is a palindrome.");
  } else {
    print("The string '$inputStr' is not a palindrome.");
  }
}
