void main() {
  String str1 = "Hello";
  String str2 = "World";

  String concatenatedStr = str1 + " " + str2;

  int length = concatenatedStr.length;

  String substring = concatenatedStr.substring(0, 5);

  String upperCaseStr = concatenatedStr.toUpperCase();

  String lowerCaseStr = concatenatedStr.toLowerCase();

  print("Concatenated String: $concatenatedStr");
  print("Length of Concatenated String: $length");
  print("Substring (0-5): $substring");
  print("Uppercase: $upperCaseStr");
  print("Lowercase: $lowerCaseStr");
}
